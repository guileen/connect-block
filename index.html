    var block = require('connect-block');
    connect(
        block('IE8'), //you can use IE, IE6, IE7, IE8, IE9, IE10 or RegExp
        function(req, res){
          res.end('You have passed.');
        }
    ).listen(3000);
If you do not browse http://localhost:3000 with IE6-IE8, you will got "You have passed", but if you do you will got the screen below.
